import "experimental"

temp = from(bucket: "telemetry")
  |> range(start: -2h)
  |> filter(fn: (r) => r._measurement == "device_telemetry")
  |> filter(fn: (r) => r._field == "temperature_c")
  |> group(columns: ["device_id"])
  |> last()
  |> keep(columns: ["device_id", "_value"])
  |> rename(columns: {_value: "Aktuelle Temperatur"})

gts = from(bucket: "telemetry")
  |> range(start: -400d)
  |> filter(fn: (r) => r._measurement == "gts_daily")
  |> filter(fn: (r) => r._field == "gts_value")
  |> group(columns: ["device_id"])
  |> last()
  |> keep(columns: ["device_id", "_value"])
  |> rename(columns: {_value: "GTS aktuelles Jahr"})

weight = from(bucket: "telemetry")
  |> range(start: -2h)
  |> filter(fn: (r) => r._measurement == "hive_telemetry")
  |> filter(fn: (r) => r._field == "weight_kg")
  |> group(columns: ["device_id", "hive_name"])
  |> last()
  |> keep(columns: ["device_id", "hive_name", "_value"])
  |> rename(columns: {_value: "Beutengewicht"})

today_first = from(bucket: "telemetry")
  |> range(start: today())
  |> filter(fn: (r) => r._measurement == "hive_telemetry")
  |> filter(fn: (r) => r._field == "weight_kg")
  |> group(columns: ["device_id", "hive_name"])
  |> first()
  |> keep(columns: ["device_id", "hive_name", "_value"])
  |> rename(columns: {_value: "today_first"})

today_last = from(bucket: "telemetry")
  |> range(start: today())
  |> filter(fn: (r) => r._measurement == "hive_telemetry")
  |> filter(fn: (r) => r._field == "weight_kg")
  |> group(columns: ["device_id", "hive_name"])
  |> last()
  |> keep(columns: ["device_id", "hive_name", "_value"])
  |> rename(columns: {_value: "today_last"})

today_join = join(tables: {f: today_first, l: today_last}, on: ["device_id", "hive_name"], method: "inner")
today_yield = today_join
  |> map(fn: (r) => ({ device_id: r.device_id, hive_name: r.hive_name, "Ertrag heute": r.today_last - r.today_first }))
  |> keep(columns: ["device_id", "hive_name", "Ertrag heute"])

yesterday_start = experimental.subDuration(from: today(), d: 1d)
yesterday_first = from(bucket: "telemetry")
  |> range(start: yesterday_start, stop: today())
  |> filter(fn: (r) => r._measurement == "hive_telemetry")
  |> filter(fn: (r) => r._field == "weight_kg")
  |> group(columns: ["device_id", "hive_name"])
  |> first()
  |> keep(columns: ["device_id", "hive_name", "_value"])
  |> rename(columns: {_value: "yesterday_first"})

yesterday_last = from(bucket: "telemetry")
  |> range(start: yesterday_start, stop: today())
  |> filter(fn: (r) => r._measurement == "hive_telemetry")
  |> filter(fn: (r) => r._field == "weight_kg")
  |> group(columns: ["device_id", "hive_name"])
  |> last()
  |> keep(columns: ["device_id", "hive_name", "_value"])
  |> rename(columns: {_value: "yesterday_last"})

yesterday_join = join(tables: {f: yesterday_first, l: yesterday_last}, on: ["device_id", "hive_name"], method: "inner")
yesterday_yield = yesterday_join
  |> map(fn: (r) => ({ device_id: r.device_id, hive_name: r.hive_name, "Ertrag gestern": r.yesterday_last - r.yesterday_first }))
  |> keep(columns: ["device_id", "hive_name", "Ertrag gestern"])

base1 = join(tables: {w: weight, t: temp}, on: ["device_id"], method: "inner")
base2 = join(tables: {b: base1, g: gts}, on: ["device_id"], method: "inner")
base3 = join(tables: {b: base2, ty: today_yield}, on: ["device_id", "hive_name"], method: "inner")
base4 = join(tables: {b: base3, yy: yesterday_yield}, on: ["device_id", "hive_name"], method: "inner")

base4
  |> keep(columns: ["device_id", "hive_name", "Aktuelle Temperatur", "GTS aktuelles Jahr", "Beutengewicht", "Ertrag heute", "Ertrag gestern"])
  |> sort(columns: ["device_id", "hive_name"])
